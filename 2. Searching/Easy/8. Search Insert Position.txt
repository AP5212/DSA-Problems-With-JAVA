https://leetcode.com/problems/search-insert-position/


class Solution {
    public int searchInsert(int[] nums, int target) {
        int n=nums.length;
        int low=0,high=n-1,mid=0;
        while(low<=high){
            mid=(low+high)/2;
            if(nums[mid]==target)
                return mid;
            else if(target<nums[mid])
                high=mid-1;
            else
                low=mid+1;
        }
        if(nums[mid]>target)
                return mid;
        return mid+1;
    }
}
